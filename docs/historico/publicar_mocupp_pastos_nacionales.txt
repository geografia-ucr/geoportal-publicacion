# --- Paso 0: Preparación local (directorio mocupp_pastos/) ---

# Archivos originales recibidos -> nombre definitivo (typeName) y título:
#   Pastos_2021_Finales                  -> cobertura_pastos_cr_2021
#        "Cobertura de pastos en Costa Rica 2021"
#   Pastos_2022_Finales                  -> cobertura_pastos_cr_2022
#        "Cobertura de pastos en Costa Rica 2022"
#   cobertura_arborea_2021_ver02         -> cobertura_arborea_pastos_cr_2021
#        "Cobertura arbórea en paisaje productivo de pastos en Costa Rica 2021"
#   cobertura_arborea_2022_ver02         -> cobertura_arborea_pastos_cr_2022
#        "Cobertura arbórea en paisaje productivo de pastos en Costa Rica 2022"
#   Perdida_Ganancia_Cobertura_2021_2022 -> perdida_ganancia_cobertura_arborea_pastos_cr_2021_2022
#        "Pérdida y ganancia de cobertura arbórea sobre paisaje productivo de pastos en Costa Rica 2021-2022"

# Verificación de CRS: las dos capas de cobertura arbórea llegaron en
# CR05 / CRTM05 (EPSG:5367) y se transformaron a CR-SIRGAS / CRTM05 (EPSG:8908)
# con la transformación oficial de EPSG "CR05 to CR-SIRGAS (1)" (Helmert,
# exactitud 0.5 m; mismos parámetros del WKT publicado por el SNIT en
# https://www.snitcr.go.cr/pdfs/utilidades/WKT_CRSIRGAS_WGS84.txt):

ogr2ogr -f "ESRI Shapefile" -s_srs EPSG:5367 -t_srs EPSG:8908 -lco ENCODING=UTF-8 \
  cobertura_arborea_pastos_cr_2021.shp cobertura_arborea_2021_ver02.shp
ogr2ogr -f "ESRI Shapefile" -s_srs EPSG:5367 -t_srs EPSG:8908 -lco ENCODING=UTF-8 \
  cobertura_arborea_pastos_cr_2022.shp cobertura_arborea_2022_ver02.shp

# Las otras 3 capas ya venían en EPSG:8908; solo se renombraron (shp, shx, dbf, prj, cpg).
# Verificación final: python3 verificar_crs.py -> las 5 capas [CORRECTO] EPSG:8908.
# Los archivos originales quedaron en mocupp_pastos/originales/ (no se subieron).

# --- Paso 1: Respaldo en el servidor ---

ssh -t -o PubkeyAuthentication=no geoportal@geoportal.ucr.ac.cr \
  "sudo bash -c 'cd /var/lib/docker/volumes/geoportal-gsdatadir/_data && \
   zip -rq /home/geoportal/geoserver-data-mocupp_pastos-20260714.zip mocupp_pastos'"

# --- Paso 2: Subir shapefiles y copiar al directorio de datos ---

scp -o PubkeyAuthentication=no \
  cobertura_pastos_cr_2021.* cobertura_pastos_cr_2022.* \
  cobertura_arborea_pastos_cr_2021.* cobertura_arborea_pastos_cr_2022.* \
  perdida_ganancia_cobertura_arborea_pastos_cr_2021_2022.* \
  geoportal@geoportal.ucr.ac.cr:/tmp/

ssh -t -o PubkeyAuthentication=no geoportal@geoportal.ucr.ac.cr \
  "sudo bash -c 'cp /tmp/cobertura_pastos_cr_20*.* /tmp/cobertura_arborea_pastos_cr_20*.* \
   /tmp/perdida_ganancia_cobertura_arborea_pastos_cr_2021_2022.* \
   /var/lib/docker/volumes/geoportal-gsdatadir/_data/mocupp_pastos/'"

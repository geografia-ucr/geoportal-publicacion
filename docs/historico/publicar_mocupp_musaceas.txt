# Registro histórico de comandos ejecutados (sanitizado). El flujo vigente es `geopub` — ver docs/flujo-publicacion.md

# =============================================================================
# Publicar mocupp_musaceas en GeoServer
# Ejecutar cada comando uno por uno desde /home/mfvargas/geoportal/publicacion
# =============================================================================

# --- Paso 1: Verificar CRS localmente ---

cd /home/mfvargas/geoportal/publicacion
python3 verificar_crs.py

# --- Paso 2: Respaldar el directorio de datos de GeoServer ---

ssh -t -o PubkeyAuthentication=no geoportal@geoportal.ucr.ac.cr "sudo zip -r /home/geoportal/bak/geoserver-data-mocupp-$(date +%Y%m%d).zip /var/lib/docker/volumes/geoportal-gsdatadir/_data/mocupp"

# Descargar respaldo a la estación de trabajo local (opcional)
scp -o PubkeyAuthentication=no geoportal@geoportal.ucr.ac.cr:/home/geoportal/bak/geoserver-data-mocupp-$(date +%Y%m%d).zip .

# --- Paso 3: Subir archivos al servidor ---

# Crear directorio temporal en el servidor
ssh -o PubkeyAuthentication=no geoportal@geoportal.ucr.ac.cr "mkdir -p /tmp/mocupp-musaceas-temp"

# Copiar shapefiles
scp -o PubkeyAuthentication=no mocupp_musaceas/* geoportal@geoportal.ucr.ac.cr:/tmp/mocupp-musaceas-temp/

# Copiar al directorio de datos de GeoServer (requiere sudo)
ssh -t -o PubkeyAuthentication=no geoportal@geoportal.ucr.ac.cr "sudo mkdir -p /var/lib/docker/volumes/geoportal-gsdatadir/_data/mocupp_musaceas && sudo cp /tmp/mocupp-musaceas-temp/* /var/lib/docker/volumes/geoportal-gsdatadir/_data/mocupp_musaceas/"

# Verificar la copia
ssh -t -o PubkeyAuthentication=no geoportal@geoportal.ucr.ac.cr "sudo ls -lh /var/lib/docker/volumes/geoportal-gsdatadir/_data/mocupp_musaceas/"

# Limpiar directorio temporal
ssh -o PubkeyAuthentication=no geoportal@geoportal.ucr.ac.cr "rm -rf /tmp/mocupp-musaceas-temp"

# --- Paso 4: Crear workspace ---

curl -s -u admin:$GEOSERVER_PASSWORD \
  -XPOST \
  -H "Content-type: application/json" \
  -d '{"workspace":{"name":"mocupp_musaceas"}}' \
  "https://geoportal.ucr.ac.cr/geoserver/rest/workspaces" | jq

# Verificar
curl -s -u admin:$GEOSERVER_PASSWORD \
  "https://geoportal.ucr.ac.cr/geoserver/rest/workspaces.json" | jq

# --- Paso 5: Crear datastore de tipo directorio de shapefiles ---

curl -s -u admin:$GEOSERVER_PASSWORD \
  -XPOST \
  -H "Content-type: application/json" \
  -d '{
    "dataStore": {
      "name": "mocupp_musaceas",
      "type": "Directory of spatial files (Shapefiles)",
      "connectionParameters": {
        "entry": [
          {"@key": "url", "$": "file:///geoserver_data/data/mocupp_musaceas"},
          {"@key": "charset", "$": "UTF-8"}
        ]
      }
    }
  }' \
  "https://geoportal.ucr.ac.cr/geoserver/rest/workspaces/mocupp_musaceas/datastores" | jq

# --- Paso 6: Publicar las capas ---

# Publicar capa musaceas_cr_2021
curl -s -u admin:$GEOSERVER_PASSWORD \
  -XPOST \
  -H "Content-type: application/json" \
  -d '{
    "featureType": {
      "name": "musaceas_cr_2021",
      "nativeCRS": "EPSG:8908",
      "srs": "EPSG:8908",
      "enabled": true
    }
  }' \
  "https://geoportal.ucr.ac.cr/geoserver/rest/workspaces/mocupp_musaceas/datastores/mocupp_musaceas/featuretypes" | jq

# Publicar capa musaceas_cr_2022
curl -s -u admin:$GEOSERVER_PASSWORD \
  -XPOST \
  -H "Content-type: application/json" \
  -d '{
    "featureType": {
      "name": "musaceas_cr_2022",
      "nativeCRS": "EPSG:8908",
      "srs": "EPSG:8908",
      "enabled": true
    }
  }' \
  "https://geoportal.ucr.ac.cr/geoserver/rest/workspaces/mocupp_musaceas/datastores/mocupp_musaceas/featuretypes" | jq

# Publicar capa perdida_ganancia_musaceas_2021_2022
curl -s -u admin:$GEOSERVER_PASSWORD \
  -XPOST \
  -H "Content-type: application/json" \
  -d '{
    "featureType": {
      "name": "perdida_ganancia_musaceas_2021_2022",
      "nativeCRS": "EPSG:8908",
      "srs": "EPSG:8908",
      "enabled": true
    }
  }' \
  "https://geoportal.ucr.ac.cr/geoserver/rest/workspaces/mocupp_musaceas/datastores/mocupp_musaceas/featuretypes" | jq

# --- Paso 7: Agregar regla en GeoFence ---
# En la interfaz web de GeoServer: Security > GeoFence Data Rules > Add new rule
# Configurar: Role=*, User=*, Service=*, Request=*, SubField=*,
#             Workspace=mocupp_musaceas, Layer=*, Access=ALLOW
# La regla se aplica de inmediato, no requiere reiniciar.

# --- Paso 8: Verificar la publicación ---

# Listar capas del workspace
curl -s -u admin:$GEOSERVER_PASSWORD \
  "https://geoportal.ucr.ac.cr/geoserver/rest/workspaces/mocupp_musaceas/layers.json" | jq

# Verificar en navegador: https://geoportal.ucr.ac.cr/geoserver/web/ -> Layer Preview
